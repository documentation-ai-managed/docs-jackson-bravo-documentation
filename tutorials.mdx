## Tutorial 3: Advanced Design Techniques

Master advanced features to create more sophisticated and interactive websites.

```javascript
function createInteractiveGallery(images) {
  const gallery = document.querySelector('.gallery');
  const modal = document.createElement('div');
  modal.className = 'modal';
  
  images.forEach((image, index) => {
    const img = document.createElement('img');
    img.src = image.src;
    img.alt = image.alt;
    img.addEventListener('click', () => {
      modal.innerHTML = `<img src="${image.src}" alt="${image.alt}">
<button class="close">&times;</button>`;
      document.body.appendChild(modal);
      modal.style.display = 'block';
    });
    gallery.appendChild(img);
  });
  
  // Close modal functionality
  modal.addEventListener('click', (e) => {
    if (e.target.classList.contains('close') || e.target === modal) {
      modal.style.display = 'none';
    }
  });
}

const portfolioImages = [
  { src: 'project1.jpg', alt: 'Brand Identity Design' },
  { src: 'project2.jpg', alt: 'Website Redesign' },
  { src: 'project3.jpg', alt: 'Mobile App UI' }
];

createInteractiveGallery(portfolioImages);
```

<Callout kind="tip">Experiment with CSS animations and transitions to add subtle interactivity to your designs.</Callout>
